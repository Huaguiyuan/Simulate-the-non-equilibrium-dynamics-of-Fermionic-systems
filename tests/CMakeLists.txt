function(add_ieompp_test TEST_NAME TEST_SOURCE)
    add_executable(${TEST_NAME} "${TEST_SOURCE}")
    if(${ieompp_ENABLE_COVERAGE})
        target_link_libraries(${TEST_NAME} gcov)
        set_source_files_properties(${TEST_SOURCE} PROPERTIES COMPILE_FLAGS "${TEST_EXTRA_FLAGS} --coverage")
    else()
        set_source_files_properties(${TEST_SOURCE} PROPERTIES COMPILE_FLAGS "${TEST_EXTRA_FLAGS}")
    endif()
    add_test(${TEST_NAME} ${TEST_NAME})
endfunction()
